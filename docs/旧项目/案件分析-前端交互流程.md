# 案件分析 - 前端交互流程

## 一、Socket 连接管理

### 1.1 连接初始化

```javascript
// stores/socket.js
connect(url = null, options = {}) {
  const socketUrl = url || this.getDefaultUrl();
  const token = localStorage.getItem('token');
  
  this.socket = io(socketUrl, {
    auth: { token },
    withCredentials: true,
    reconnection: true,
    reconnectionAttempts: 5,
    reconnectionDelay: 1000,
    reconnectionDelayMax: 5000,
    timeout: 20000,
    ...options
  });
  
  this.bindEventListeners();
}
```

### 1.2 事件监听绑定

```javascript
bindEventListeners() {
  // 连接成功
  this.socket.on('connect', () => {
    this.isConnected = true;
    this.connectionId = this.socket.id;
    this.rejoinRooms();
    this.processMessageQueue();
  });
  
  // 断开连接
  this.socket.on('disconnect', (reason) => {
    this.isConnected = false;
  });
  
  // 连接错误
  this.socket.on('connect_error', (error) => {
    this.isError = true;
    this.lastError = error;
  });
}
```

## 二、创建案件流程

### 2.1 流程图

```
用户上传材料
      │
      ▼
┌─────────────────┐
│ 生成 sessionId  │  emit: generateCaseSessionId
└─────────────────┘
      │
      ▼
┌─────────────────┐
│   创建案件      │  emit: createCase
└─────────────────┘
      │
      ▼
┌─────────────────┐
│ 接收材料消息    │  on: case:analysis:message (addMaterial)
└─────────────────┘
      │
      ▼
┌─────────────────┐
│ 接收等待消息    │  on: case:analysis:message (wait)
└─────────────────┘
      │
      ▼
┌─────────────────┐
│ 接收分析过程    │  on: case:analysis:message (reasoning/content)
└─────────────────┘
      │
      ▼

│ 接收完成消息    │  on: case:analysis:message (finished)
└─────────────────┘
      │
      ▼
┌─────────────────┐
│ 接收确认消息    │  on: case:analysis:message (userConfirm)
└─────────────────┘
```

### 2.2 创建案件代码示例

```javascript
// 生成会话 ID

  }

// 创建案件

```

### 2.3 监听分析消息

```javascript
// 监听分析消息
socketStore.on('case:analysis:message', (message) => {
  const { messageType, data, id, sessionId } = message;
  
  switch (messageType) {
    case 'addMaterial':
      // 添加材料到列表
      analysisAgentStore.setMaterialList({
        id: data.materialId,
        name: data.name,
        type: data.materialType,
        // ...
      });
      break;
      
    case 'wait':
      // 显示等待状态
      analysisAgentStore.addOrUpdateMessage({
        id,
        messageType: 'wait',
        message: data
      });
      break;
      
    case 'reasoning':
      // 显示推理过程
      analysisAgentStore.addOrUpdateMessage({
        id,
        messageType: 'reasoning',
        message: { reasoning: data.reasoning }
      });
      break;
      
    case 'finished':
      // 更新案件信息
      analysisAgentStore.title = data.title;
      analysisAgentStore.plaintiff = data.plaintiff;
      analysisAgentStore.defendant = data.defendant;
      break;
      
    case 'userConfirm':
      // 显示确认表单
      analysisAgentStore.addOrUpdateMessage({
        id,
        messageType: 'userConfirm',
        message: data
      });
      break;
  }
});
```

## 五、分析过程展示

### 5.1 消息类型处理

```javascript
// 处理分析过程消息
const handleAnalysisMessage = (message) => {
  const { messageType, data, id } = message;
  
  switch (messageType) {
    case 'taskStart':
      // 任务开始
      updateTaskStatus(data.moduleName, 'processing');
      break;
      
    case 'reasoning':
      // 推理过程（流式）
      appendReasoning(id, data.reasoning);
      break;
      
    case 'reasoningEnd':
      // 推理结束
      finalizeReasoning(id, data.reasoning);
      break;
      
    case 'toolStart':
      // 工具调用开始
      showToolCall(id, data.tool_name, data.tool_input);
      break;
      
    case 'toolEnd':
      // 工具调用结束
      updateToolResult(id, data.tool_output);
      break;
      
    case 'content':
      // 内容输出（流式）
      appendContent(id, data.content);
      break;
      
    case 'contentEnd':
      // 内容输出结束
      finalizeContent(id, data.content);
      break;
      
    case 'taskEnd':
      // 任务结束
      updateTaskStatus(data.moduleName, 'completed');
      saveAnalysisResult(data.moduleName, data.content);
      break;
      
    case 'task':
      // 任务状态更新
      updateAllTaskStatus(data);
      break;
  }
};
```

### 5.2 流式内容展示

```javascript
// 流式内容追加
const appendContent = (messageId, content) => {
  const message = messages.value.find(m => m.id === messageId);
  if (message) {
    message.content = (message.content || '') + content;
  } else {
    messages.value.push({
      id: messageId,
      messageType: 'content',
      content: content
    });
  }
};

// 内容最终化
const finalizeContent = (messageId, content) => {
  const message = messages.value.find(m => m.id === messageId);
  if (message) {
    message.content = content;
    message.completed = true;
  }
};
```

## 十、性能优化

### 10.1 消息队列

```javascript
// 离线消息队列
emit(event, data, callback) {
  if (!this.isConnected) {
    this.messageQueue.push({ type: 'emit', event, data, callback });
    return false;
  }
  // ...
}

// 重连后处理队列
processMessageQueue() {
  this.messageQueue.forEach(message => {
    this.emit(message.event, message.data, message.callback);
  });
  this.messageQueue = [];
}
```

### 10.2 事件监听器缓存

```javascript
// 缓存监听器
on(event, handler) {
  if (!this.eventListeners.has(event)) {
    this.eventListeners.set(event, []);
  }
  this.eventListeners.get(event).push(handler);
  
  if (this.socket) {
    this.socket.on(event, handler);
  }
}

// 重连后恢复监听
bindCachedListeners() {
  this.eventListeners.forEach((handlers, event) => {
    handlers.forEach(handler => {
      this.socket.on(event, handler);
    });
  });
}
```
